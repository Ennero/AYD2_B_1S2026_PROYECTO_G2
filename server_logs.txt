
    IF NOT FOUND THEN
        RAISE EXCEPTION 'El contrato % no existe.', NEW.CONTRACT_ID;
    END IF;

    IF V_CONTRACT_STATUS <> 'VIGENTE' THEN
        RAISE EXCEPTION 'No se puede crear la orden porque el contrato no esta vigente.';
    END IF;

    IF CURRENT_DATE < V_START_DATE OR CURRENT_DATE > V_END_DATE THEN
        RAISE EXCEPTION 'No se puede crear la orden porque el contrato esta fuera de vigencia.';
    END IF;

    IF V_IS_BLOCKED THEN
        RAISE EXCEPTION 'No se puede crear la orden porque el cliente esta bloqueado: %', COALESCE(V_BLOCK_REASON, 'sin motivo especificado');
    END IF;

    SELECT
        COALESCE(SUM(I.TOTAL_AMOUNT) FILTER (WHERE I.STATUS <> 'RECHAZADA'), 0),
        COALESCE(SUM(P.AMOUNT) FILTER (WHERE P.STATUS = 'APROBADO'), 0),
        COUNT(*) FILTER (
            WHERE I.STATUS <> 'PAGADA'
              AND I.STATUS <> 'RECHAZADA'
              AND I.DUE_DATE < CURRENT_DATE
        )
    INTO
        V_INVOICED_TOTAL,
        V_PAID_TOTAL,
        V_OVERDUE_INVOICES
    FROM CONTRACTS CT
    LEFT JOIN ORDERS O
      ON O.CONTRACT_ID = CT.CONTRACT_ID
    LEFT JOIN INVOICES I
      ON I.ORDER_ID = O.ORDER_ID
    LEFT JOIN PAYMENTS P
      ON P.INVOICE_ID = I.INVOICE_ID
    WHERE CT.CLIENT_ID = V_CLIENT_ID;

    IF V_OVERDUE_INVOICES > 0 THEN
        RAISE EXCEPTION 'No se puede crear la orden porque el cliente tiene facturas vencidas.';
    END IF;

    IF V_CONTRACT_CREDIT_LIMIT - (V_INVOICED_TOTAL - V_PAID_TOTAL) < 0 THEN
        RAISE EXCEPTION 'No se puede crear la orden porque el cliente excedio su limite de credito.';
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION POPULATE_INVOICE_FROM_ORDER()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    V_CLIENT_ID UUID;
    V_CLIENT_NAME VARCHAR(180);
    V_CLIENT_NIT VARCHAR(20);
    V_CLIENT_ADDRESS TEXT;
    V_ORDER_NUMBER VARCHAR(40);
    V_TOTAL NUMERIC(14, 2);
    V_SUBTOTAL NUMERIC(14, 2);
    V_TAX NUMERIC(14, 2);
    V_TERM SMALLINT;
BEGIN
    SELECT
        C.CLIENT_ID,
        C.LEGAL_NAME,
        C.NIT,
        C.TAX_ADDRESS,
        O.ORDER_NUMBER,
        O.TOTAL_AMOUNT,
        O.SUBTOTAL_AMOUNT,
        O.TAX_AMOUNT,
        CT.PAYMENT_TERM_DAYS
      INTO
        V_CLIENT_ID,
        V_CLIENT_NAME,
        V_CLIENT_NIT,
        V_CLIENT_ADDRESS,
        V_ORDER_NUMBER,
        V_TOTAL,
        V_SUBTOTAL,
        V_TAX,
        V_TERM
    FROM ORDERS O
    JOIN CONTRACTS CT
      ON CT.CONTRACT_ID = O.CONTRACT_ID
    JOIN CLIENTS C
      ON C.CLIENT_ID = CT.CLIENT_ID
    WHERE O.ORDER_ID = NEW.ORDER_ID;

    IF NEW.CLIENT_ID IS NULL THEN
        NEW.CLIENT_ID = V_CLIENT_ID;
    END IF;

    IF NEW.CLIENT_NAME IS NULL OR NEW.CLIENT_NAME = '' THEN
        NEW.CLIENT_NAME = V_CLIENT_NAME;
    END IF;

    IF NEW.CLIENT_NIT IS NULL OR NEW.CLIENT_NIT = '' THEN
        NEW.CLIENT_NIT = V_CLIENT_NIT;
    END IF;

    IF NEW.CLIENT_ADDRESS IS NULL OR NEW.CLIENT_ADDRESS = '' THEN
        NEW.CLIENT_ADDRESS = V_CLIENT_ADDRESS;
    END IF;

    IF NEW.SERVICE_DESCRIPTION IS NULL OR NEW.SERVICE_DESCRIPTION = '' THEN
        NEW.SERVICE_DESCRIPTION = 'SERVICIO LOGISTICO DE LA ORDEN ' || V_ORDER_NUMBER;
    END IF;

    IF COALESCE(NEW.SUBTOTAL_AMOUNT, 0) = 0 THEN
        NEW.SUBTOTAL_AMOUNT = V_SUBTOTAL;
    END IF;

    IF COALESCE(NEW.TAX_AMOUNT, 0) = 0 THEN
        NEW.TAX_AMOUNT = V_TAX;
    END IF;

    IF COALESCE(NEW.TOTAL_AMOUNT, 0) = 0 THEN
        NEW.TOTAL_AMOUNT = V_TOTAL;
    END IF;

    IF NEW.DUE_DATE IS NULL THEN
        NEW.DUE_DATE = (COALESCE(NEW.ISSUE_DATE, NOW())::DATE + COALESCE(V_TERM, 30));
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION VALIDATE_PAYMENT_AMOUNT()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    V_TOTAL NUMERIC(14, 2);
        V_APPROVED_PAYMENT_ID UUID;
        V_INVOICE_STATUS INVOICE_STATUS;
BEGIN
        SELECT TOTAL_AMOUNT, STATUS
            INTO V_TOTAL, V_INVOICE_STATUS
    FROM INVOICES
    WHERE INVOICE_ID = NEW.INVOICE_ID;

    IF V_TOTAL IS NULL THEN
        RAISE EXCEPTION 'La factura % no existe.', NEW.INVOICE_ID;
    END IF;

        IF V_INVOICE_STATUS = 'PAGADA' THEN
                RAISE EXCEPTION 'La factura % ya fue pagada.', NEW.INVOICE_ID;
        END IF;

        SELECT PAYMENT_ID
            INTO V_APPROVED_PAYMENT_ID
        FROM PAYMENTS
        WHERE INVOICE_ID = NEW.INVOICE_ID
            AND STATUS = 'APROBADO'
            AND PAYMENT_ID <> NEW.PAYMENT_ID
        LIMIT 1;

        IF V_APPROVED_PAYMENT_ID IS NOT NULL THEN
                RAISE EXCEPTION 'La factura % ya tiene un pago aprobado.', NEW.INVOICE_ID;
        END IF;

    IF NEW.AMOUNT <> V_TOTAL THEN
        RAISE EXCEPTION 'El pago % debe coincidir exactamente con el total de la factura %.', NEW.AMOUNT, V_TOTAL;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION SYNC_INVOICE_PAYMENT()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    IF NEW.STATUS = 'APROBADO' THEN
        UPDATE INVOICES
        SET STATUS = 'PAGADA'
        WHERE INVOICE_ID = NEW.INVOICE_ID;
    END IF;

    RETURN NEW;
END;
$$;

CREATE OR REPLACE VIEW V_CLIENT_BALANCES AS
WITH INVOICE_TOTALS AS (
    SELECT
        C.CLIENT_ID,
        COALESCE(SUM(I.TOTAL_AMOUNT) FILTER (WHERE I.STATUS <> 'RECHAZADA'), 0) AS INVOICED_TOTAL,
        COUNT(*) FILTER (
            WHERE I.STATUS <> 'PAGADA'
              AND I.STATUS <> 'RECHAZADA'
              AND I.DUE_DATE < CURRENT_DATE
        ) AS OVERDUE_INVOICES
    FROM CLIENTS C
    LEFT JOIN CONTRACTS CT
        ON CT.CLIENT_ID = C.CLIENT_ID
    LEFT JOIN ORDERS O
        ON O.CONTRACT_ID = CT.CONTRACT_ID
    LEFT JOIN INVOICES I
        ON I.ORDER_ID = O.ORDER_ID
    GROUP BY C.CLIENT_ID
),
PAYMENTS_TOTALS AS (
    SELECT
        C.CLIENT_ID,
        COALESCE(SUM(P.AMOUNT) FILTER (WHERE P.STATUS = 'APROBADO'), 0) AS PAID_TOTAL
    FROM CLIENTS C
    LEFT JOIN CONTRACTS CT
        ON CT.CLIENT_ID = C.CLIENT_ID
    LEFT JOIN ORDERS O
        ON O.CONTRACT_ID = CT.CONTRACT_ID
    LEFT JOIN INVOICES I
        ON I.ORDER_ID = O.ORDER_ID
    LEFT JOIN PAYMENTS P
        ON P.INVOICE_ID = I.INVOICE_ID
    GROUP BY C.CLIENT_ID
)
SELECT
    C.CLIENT_ID,
    C.CLIENT_CODE,
    C.LEGAL_NAME,
    C.CREDIT_LIMIT,
    COALESCE(IT.INVOICED_TOTAL, 0) - COALESCE(PT.PAID_TOTAL, 0) AS OUTSTANDING_BALANCE,
    C.CREDIT_LIMIT - (COALESCE(IT.INVOICED_TOTAL, 0) - COALESCE(PT.PAID_TOTAL, 0)) AS AVAILABLE_CREDIT,
    COALESCE(IT.OVERDUE_INVOICES, 0) AS OVERDUE_INVOICES,
    (
        C.IS_BLOCKED
        OR COALESCE(IT.OVERDUE_INVOICES, 0) > 0
        OR C.CREDIT_LIMIT - (COALESCE(IT.INVOICED_TOTAL, 0) - COALESCE(PT.PAID_TOTAL, 0)) < 0
    ) AS SHOULD_BLOCK_ORDERS
FROM CLIENTS C
LEFT JOIN INVOICE_TOTALS IT
    ON IT.CLIENT_ID = C.CLIENT_ID
LEFT JOIN PAYMENTS_TOTALS PT
    ON PT.CLIENT_ID = C.CLIENT_ID;

CREATE OR REPLACE VIEW V_DAILY_OPERATIONS AS
SELECT
    B.BRANCH_ID,
    B.BRANCH_NAME,
    DATE(O.REQUESTED_AT) AS OPERATION_DATE,
    COUNT(*) AS TOTAL_ORDERS,
    COUNT(*) FILTER (WHERE O.STATUS = 'ENTREGADA') AS DELIVERED_ORDERS,
    COALESCE(SUM(I.TOTAL_AMOUNT) FILTER (WHERE I.STATUS <> 'RECHAZADA'), 0) AS INVOICED_TOTAL,
    COUNT(*) FILTER (WHERE I.STATUS = 'PAGADA') AS PAID_INVOICES
FROM BRANCHES B
LEFT JOIN ORDERS O
    ON O.BRANCH_ID = B.BRANCH_ID
LEFT JOIN INVOICES I
    ON I.ORDER_ID = O.ORDER_ID
GROUP BY B.BRANCH_ID, B.BRANCH_NAME, DATE(O.REQUESTED_AT);

CREATE OR REPLACE VIEW V_CONTRACT_PROFITABILITY AS
SELECT
    CT.CONTRACT_ID,
    CT.CONTRACT_NUMBER,
    CT.CLIENT_ID,
    COALESCE(SUM(I.TOTAL_AMOUNT) FILTER (WHERE I.STATUS <> 'RECHAZADA'), 0) AS REVENUE_AMOUNT,
    COALESCE(SUM(O.FUEL_COST + O.VIATICS_COST + O.MAINTENANCE_COST), 0) AS OPERATING_COST_AMOUNT,
    COALESCE(SUM(I.TOTAL_AMOUNT) FILTER (WHERE I.STATUS <> 'RECHAZADA'), 0)
        - COALESCE(SUM(O.FUEL_COST + O.VIATICS_COST + O.MAINTENANCE_COST), 0) AS GROSS_MARGIN_AMOUNT
FROM CONTRACTS CT
LEFT JOIN ORDERS O
    ON O.CONTRACT_ID = CT.CONTRACT_ID
LEFT JOIN INVOICES I
    ON I.ORDER_ID = O.ORDER_ID
GROUP BY CT.CONTRACT_ID, CT.CONTRACT_NUMBER, CT.CLIENT_ID;
